166 if (debugOutput) {
167 std::cout << "Node " << address << " at time " << simTime() << ": has no packets in queue, ending transmission window" << endl;
168 }
169 currentState = KIWAN_IDLE;
170 scheduleAt(simTime() + ttwIdle, this->eventScheduleTransmission);
171 return;
172 }
173 } else if (!currentTxFrame) {
174 // If somehow we don't have a frame for subsequent windows
175 if (debugOutput) {
176 std::cout << "Node " << address << " at time " << simTime() << ": has no current frame for window #" << (currentTwCount + 1) << ", ending transmission" << endl;
177 }
178 currentState = KIWAN_IDLE;
179 scheduleAt(simTime() + ttwIdle, this->eventScheduleTransmission);
180 return;
181 }
182
183 // Schedule the first transmission after ttxStart delay randomly within ttxStart_min and ttxStart_max
184 ttxStart = uniform(ttxStart_min, ttxStart_max);
185 if (debugOutput) {
186 std::cout << "Node " << address << " at time " << simTime() << ": scheduling first transmission in window #" << (currentTwCount + 1) << " after " << ttxStart << " seconds" << endl;
187 }
188
189 scheduleAt(simTime() + ttxStart, this->eventTransmitData);
190
191 // Schedule end of this transmission window
192 scheduleAt(simTime() + ttwDuration, this->eventTransmitWindowEnd);
193
194 currentTwCount++;
195 } else {
196 // All transmission windows completed
197 currentState = KIWAN_IDLE;
198 scheduleAt(simTime() + ttwIdle, this->eventScheduleTransmission);
199 if (debugOutput) {
200 std::cout << "Node " << address << " at time " << simTime() << ": completed all transmission windows, returning to IDLE" << endl;
201 }
202 }
