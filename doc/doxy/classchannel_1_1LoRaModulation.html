36 // Valid for IEEE 802.15.4 2.45 GHz OQPSK modulation
37 // Following formula is defined in IEEE 802.15.4 standard, please check the
38 // 2006 standard, page 268, section E.4.1.8 Bit error rate (BER)
39 // calculations, formula 7). Here you can see that the factor of 20.0 is correct ;).
40 const double dSNRFct = 20.0 *snir *bandwidth.get() / bitrate.get(); // TODO is this correct?
41 double dSumK = 0;
42 int k = 2;
43
44 /* following loop was optimized by using n_choose_k symmetries
45 for (k=2; k <= 16; ++k) {
46 dSumK += pow(-1.0, k) * n_choose_k(16, k) * exp(dSNRFct * (1.0 / k - 1.0));
47 }
48 */
49
50 // n_choose_k(16, k) == n_choose_k(16, 16-k)
51 for ( ; k < 8; k += 2) {
52 // k will be 2, 4, 6 (symmetric values: 14, 12, 10)
53 dSumK += math::n_choose_k(16, k) * (exp(dSNRFct * (1.0 / k - 1.0)) + exp(dSNRFct * (1.0 / (16 - k) - 1.0)));
54 }
55
56 // for k = 8 (which does not have a symmetric value)
57 k = 8;
58 dSumK += math::n_choose_k(16, k) * exp(dSNRFct * (1.0 / k - 1.0));
59 for (k = 3; k < 8; k += 2) {
60 // k will be 3, 5, 7 (symmetric values: 13, 11, 9)
61 dSumK -= math::n_choose_k(16, k) * (exp(dSNRFct * (1.0 / k - 1.0)) + exp(dSNRFct * (1.0 / (16 - k) - 1.0)));
62 }
63
64 // for k = 15 (because of missing k=1 value)
65 k = 15;
66 dSumK -= math::n_choose_k(16, k) * exp(dSNRFct * (1.0 / k - 1.0));
67
68 // for k = 16 (because of missing k=0 value)
69 k = 16;
70 dSumK += math::n_choose_k(16, k) * exp(dSNRFct * (1.0 / k - 1.0));
